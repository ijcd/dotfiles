# devshell Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Shell function + templates for quick nix dev shells across 8 ecosystems

**Architecture:** Plain `flake.nix` template files per ecosystem in `~/.config/devshell-templates/`, a shared `.envrc`, and a single `devshell` zsh function with `init`/`run`/`list` subcommands. No extra deps beyond nix + direnv.

**Tech Stack:** Nix flakes, direnv, zsh

### Task 2: Elixir template

**Files:**
- Create: `dot_config/devshell-templates/elixir/flake.nix`

- [ ] **Step 1: Create the elixir flake.nix**

```nix
{
  description = "Elixir dev shell";

  inputs = {
    nixpkgs.url = "github:nixos/nixpkgs/nixpkgs-unstable";
  };

  outputs = { nixpkgs, ... }:
    let
      forAllSystems = nixpkgs.lib.genAttrs [ "x86_64-darwin" "aarch64-darwin" "x86_64-linux" "aarch64-linux" ];
    in {
      devShells = forAllSystems (system:
        let
          pkgs = nixpkgs.legacyPackages.${system};
        in {
          default = pkgs.mkShell {
            buildInputs = with pkgs; [
              elixir
              erlang
              hex
              rebar3
              postgresql
            ] ++ pkgs.lib.optionals pkgs.stdenv.isLinux [
              inotify-tools
            ] ++ pkgs.lib.optionals pkgs.stdenv.isDarwin [
              darwin.apple_sdk.frameworks.CoreServices
              darwin.apple_sdk.frameworks.CoreFoundation
            ];

            shellHook = ''
              export LANG=en_US.UTF-8
              export ERL_AFLAGS="-kernel shell_history enabled"
              export MIX_HOME="$PWD/.mix"
              export HEX_HOME="$PWD/.hex"
              mkdir -p "$MIX_HOME" "$HEX_HOME"
            '';
          };
        });
    };
}
```

### Task 10: `devshell` zsh function

**Files:**
- Create: `dot_config/zsh/functions/devshell`
- Modify: `dot_config/zsh/interactive.zsh` (add source line)

- [ ] **Step 1: Create the devshell function**

```zsh
# devshell - Quick nix dev shells from templates
#
# Usage:
#   devshell init <ecosystem>   Scaffold flake.nix + .envrc into cwd
#   devshell run <ecosystem>    Ephemeral interactive shell
#   devshell list               Show available ecosystems

function devshell() {
  local templates_dir="${XDG_CONFIG_HOME:-$HOME/.config}/devshell-templates"

  case "$1" in
    init)
      local ecosystem="$2"
      if [[ -z "$ecosystem" ]]; then
        echo "Usage: devshell init <ecosystem>" >&2
        echo "Run 'devshell list' to see available ecosystems" >&2
        return 1
      fi

      local template_dir="$templates_dir/$ecosystem"
      if [[ ! -d "$template_dir" ]]; then
        echo "Unknown ecosystem: $ecosystem" >&2
        echo "Run 'devshell list' to see available ecosystems" >&2
        return 1
      fi

      if [[ -f flake.nix ]]; then
        echo "flake.nix already exists in current directory" >&2
        return 1
      fi

      cp "$template_dir/flake.nix" ./flake.nix
      echo "Created flake.nix ($ecosystem)"

      if [[ ! -f .envrc ]]; then
        cp "$templates_dir/envrc" ./.envrc
        echo "Created .envrc"
      else
        echo ".envrc already exists, skipping"
      fi

      if (( $+commands[direnv] )); then
        direnv allow
        echo "Ran direnv allow"
      fi

      echo "Done. Run 'nix develop' or cd out and back in for direnv."
      ;;

    run)
      local ecosystem="$2"
      if [[ -z "$ecosystem" ]]; then
        echo "Usage: devshell run <ecosystem>" >&2
        echo "Run 'devshell list' to see available ecosystems" >&2
        return 1
      fi

      local pkgs
      case "$ecosystem" in
        elixir)     pkgs=(elixir erlang hex rebar3) ;;
        go)         pkgs=(go gopls delve golangci-lint) ;;
        rust)       pkgs=(rustc cargo rustfmt clippy rust-analyzer) ;;
        python)     pkgs=(python3 black ruff mypy) ;;
        ruby)       pkgs=(ruby bundler) ;;
        typescript) pkgs=(nodejs corepack) ;;
        zig)        pkgs=(zig zls) ;;
        java)       pkgs=(jdk maven gradle) ;;
        *)
          echo "Unknown ecosystem: $ecosystem" >&2
          echo "Run 'devshell list' to see available ecosystems" >&2
          return 1
          ;;
      esac

      local nix_args=()
      for pkg in "${pkgs[@]}"; do
        nix_args+=("nixpkgs#$pkg")
      done

      echo "Entering $ecosystem shell (ephemeral)..."
      nix shell "${nix_args[@]}" -c zsh
      ;;

    list)
      echo "Available ecosystems:"
      for dir in "$templates_dir"/*/; do
        [[ -d "$dir" ]] && echo "  $(basename "$dir")"
      done
      ;;

    *)
      echo "Usage: devshell <command> [args]" >&2
      echo "" >&2
      echo "Commands:" >&2
      echo "  init <ecosystem>   Scaffold flake.nix + .envrc into cwd" >&2
      echo "  run <ecosystem>    Ephemeral interactive shell" >&2
      echo "  list               Show available ecosystems" >&2
      return 1
      ;;
  esac
}
```
